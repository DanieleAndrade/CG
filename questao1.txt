
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <GLFW/glut.h>
#include <Windows.h>
#include <time.h> 
using namespace std;


void init(void) {
	glClearColor(1.0, 1.0, 1.0, 0);
	glMatrixMode(GL_PROJECTION);
	gluOrtho2D(0, 1700, 0, 800);
}

void geraPontos() {
	glColor3f(1.0, 0.0, 0.0);
	glPolygonMode(GL_FRONT, GL_FILL);
	glBegin(GL_TRIANGLES);
	// 1o TRIANGULO
	glVertex2i(200, 300);
	glVertex2i(500, 300);
	glEdgeFlag(GL_FALSE);
	glVertex2i(350, 200);
	// 2o TRIANGULO
	glVertex2i(200, 300);
	glEdgeFlag(GL_TRUE);
	glVertex2i(285, 350);
	glVertex2i(350, 300);
	// 3o TRIANGULO
	glEdgeFlag(GL_TRUE);
	glVertex2i(350, 300);
	glVertex2i(500, 300);
	glVertex2i(450, 350);
	glEdgeFlag(GL_FALSE);

	// Tri√¢ngulo de cima
	glColor3f(0.0, 0.0, 1.0);
	glVertex2i(290, 350);
	glVertex2i(440, 350);
	glVertex2i(350, 305);

	glEnd();

	glFlush();

	
}

void lineSegment(void) {

	glClear(GL_COLOR_BUFFER_BIT);
	geraPontos();
	
}

// Programa Principal 
int main(int argc, char** argv)
{
	glutInit(&argc, argv);

	glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);

	glutInitWindowPosition(50, 100);

	glutInitWindowSize(400, 300);
	glutCreateWindow("Questao 1 - Daniele");
	init();
	glutDisplayFunc(lineSegment);
	glutMainLoop();

	return 0;
}
